    <script>

        function calculateCost() {

            let length = parseFloat(document.getElementById('length').value);

            let breadth = parseFloat(document.getElementById('breadth').value);

            let rate = 9.19;


            if (isNaN(length) || isNaN(breadth) || length <= 0 || breadth <= 0) {

                document.getElementById('materialCost').innerText = "$0.00";

                document.getElementById('monthlyPayment').innerText = "$0";

                return;

            }

            

            let materialCost = length * breadth * rate;

            let monthlyPayment = Math.floor(materialCost / 12);

            

            document.getElementById('materialCost').innerText = `$${materialCost.toFixed(2)}`;

            document.getElementById('monthlyPayment').innerText = `$${monthlyPayment}`;

        }

    </script>
